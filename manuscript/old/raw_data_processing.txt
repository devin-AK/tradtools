WORK_DIR=/work/trad/

# Download reference genomes
cd $WORK_DIR
mkdir -p genome
cd genome
wget https://hgdownload.soe.ucsc.edu/goldenPath/hg19/bigZips/hg19.fa.gz
wget https://github.com/devin-AK/small_data/blob/92f44ae07c190e5b01ca1661c26a9e804e043c27/tradtools/example_data/lambda.fa.gz
wget https://github.com/devin-AK/small_data/blob/92f44ae07c190e5b01ca1661c26a9e804e043c27/tradtools/example_data/hg19-blacklist.v2.bed.gz
cd ..

# Run the tradtools pipeline to process raw Fastq data
nextflow run devin-ak/tradtools --R1 fastq/Undetermined_HT7VYDSXX_L2_1.fq.gz \
                                --R2 fastq/Undetermined_HT7VYDSXX_L2_2.fq.gz \
                                --genome1 genome/hg19.fa.gz \
                                --genome2 genome/lambda.fa.gz \
                                --blacklist1 genome/hg19-blacklist.v2.bed.gz \
                                --adapters adapters.fa
